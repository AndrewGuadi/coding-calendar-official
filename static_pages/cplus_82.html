std::vector::push_back

The `push_back` method in C++ is used to add a new element at the end of a vector. It increases the size of the vector by one and initializes the new element with the provided value. If the vector's capacity is reached, it automatically allocates more memory to accommodate the new element. This method is particularly useful for dynamically managing a collection of data that can grow as needed.

from ctypes import c_int, POINTER, cast

class Vector:
    def __init__(self):
        self._size = 0
        self._capacity = 1
        self._data = (c_int * self._capacity)()

    def push_back(self, value):
        if self._size >= self._capacity:
            self._resize(self._capacity * 2)
        self._data[self._size] = value
        self._size += 1

    def _resize(self, new_capacity):
        new_data = (c_int * new_capacity)()
        for i in range(self._size):
            new_data[i] = self._data[i]
        self._data = new_data
        self._capacity = new_capacity

    def get_elements(self):
        return [self._data[i] for i in range(self._size)]

vector = Vector()
vector.push_back(1)
vector.push_back(2)
vector.push_back(3)
print(vector.get_elements())  # Output: [1, 2, 3]
class DynamicList:
    def __init__(self):
        self.list = []

    def push_back(self, value):
        self.list.append(value)

    def display(self):
        return self.list

my_list = DynamicList()
my_list.push_back('hello')
my_list.push_back('world')
print(my_list.display())  # Output: ['hello', 'world']
class StringContainer:
    def __init__(self):
        self.strings = []

    def push_back(self, value):
        self.strings.append(value)

    def get_all_strings(self):
        return ', '.join(self.strings)

container = StringContainer()
container.push_back('apple')
container.push_back('banana')
container.push_back('cherry')
print(container.get_all_strings())  # Output: 'apple, banana, cherry'
