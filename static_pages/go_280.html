close

The 'close' method in Go is used to close an open resource, such as a file or a network connection. Closing a resource is essential for freeing up system resources and ensuring that all buffered data is flushed and any necessary cleanup is performed. After a resource is closed, any further operations on it will result in an error.

import 'os'

// Open a file and defer its closing
file, err := os.Open("example.txt")
if err != nil {
    panic(err)
}
defer file.Close()

// Continue to use the file
// Reading or writing operations here will go before the file is closed
import 'net'

// Establish a TCP connection
conn, err := net.Dial("tcp", "golang.org:80")
if err != nil {
    panic(err)
}

defer conn.Close() // Ensure connection is closed after usage

// Use the connection for sending requests
import 'database/sql'
import _ "github.com/lib/pq" // PostgreSQL driver

// Connect to the database
db, err := sql.Open("postgres", "user=username dbname=mydb sslmode=disable")
if err != nil {
    log.Fatal(err)
}

defer db.Close() // Close the database connection

// Perform database operations here
