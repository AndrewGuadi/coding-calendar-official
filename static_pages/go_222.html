close

In Go, the 'close' method is used to close a channel, which prevents any further sends on that channel and allows any blocking receives to proceed. It is essential to ensure that you do not try to close a channel multiple times or send values on a closed channel, as this will result in a panic. Closing a channel is a signal to receivers that no more values will be sent on that channel, and they can complete their processing accordingly.

import asyncio

async def producer(queue):
    for i in range(3):
        await queue.put(i)
        print(f'Produced {i}')
    await queue.put(None)  # to signal the consumer

async def consumer(queue):
    while True:
        item = await queue.get()
        if item is None:
            break
        print(f'Consumed {item}')

async def main():
    queue = asyncio.Queue()
    await asyncio.gather(producer(queue), consumer(queue))

asyncio.run(main())
import asyncio

async def producer(channel):
    for i in range(5):
        await channel.put(i)
        print(f'Sent {i}')
    channel.close()  # closing the channel after sending all data

async def consumer(channel):
    while True:
        item = await channel.get()
        if item is None:
            break
        print(f'Received {item}')

async def main():
    channel = asyncio.Queue()
    await asyncio.gather(producer(channel), consumer(channel))

asyncio.run(main())
import asyncio

async def sender(channel):
    for message in ['Hello', 'World', 'Stop']:
        await channel.put(message)
        print(f'Sent: {message}')
    channel.close()  

async def receiver(channel):
    while True:
        try:
            item = await channel.get()
            if item == 'Stop':
                break
            print(f'Received: {item}')
        except asyncio.QueueEmpty:
            break

async def main():
    channel = asyncio.Queue()
    await asyncio.gather(sender(channel), receiver(channel))

asyncio.run(main())
