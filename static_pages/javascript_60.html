charAt

The 'charAt' method in JavaScript is used to retrieve a character at a specified index from a string. Strings in JavaScript are zero-indexed, meaning that the first character has an index of 0. If the specified index is out of bounds (negative or greater than the length of the string), 'charAt' will return an empty string.

let str = 'Hello World';
console.log(str.charAt(0)); // Output: 'H'
console.log(str.charAt(4)); // Output: 'o'
console.log(str.charAt(10)); // Output: 'd'
console.log(str.charAt(11)); // Output: '' (empty string)
let name = 'OpenAI';
console.log(name.charAt(1)); // Output: 'p'
console.log(name.charAt(2)); // Output: 'e'
console.log(name.charAt(-1)); // Output: '' (empty string)
console.log(name.charAt(6)); // Output: '' (empty string)
let message = 'JavaScript';
for (let i = 0; i < message.length; i++) {
    console.log(`Character at index ${i}: ${message.charAt(i)}`);
}
