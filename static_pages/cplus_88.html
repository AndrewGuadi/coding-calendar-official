std::vector::push_back

The `std::vector::push_back` method in C++ is used to add a new element at the end of a vector. This method allows for dynamic resizing of a vector when adding elements, handling memory allocation accordingly. If the vector is full, it automatically increases its capacity to accommodate the new element. The operation runs in constant average time, making it efficient for appending elements.

// Example of adding elements to a vector using push_back
#include 
#include 

int main() {
    std::vector pythonExamples;
    
    // Adding Python code snippets to the vector
    pythonExamples.push_back("print('Hello, World!')");
    pythonExamples.push_back("for i in range(5): print(i)");
    pythonExamples.push_back("def add(a, b): return a + b");
    
    // Display the contents of the vector
    for (const std::string &code : pythonExamples) {
        std::cout << code << std::endl;
    }
    return 0;
}

struct User {
    std::string name;
    int age;
};

int main() {
    std::vector userVector;
    
    // Adding users to the vector
    userVector.push_back({"Alice", 30});
    userVector.push_back({"Bob", 25});
    userVector.push_back({"Charlie", 35});
    
    // Displaying user details
    for (const User &user : userVector) {
        std::cout << "Name: " << user.name << ", Age: " << user.age << std::endl;
    }
    return 0;
}
