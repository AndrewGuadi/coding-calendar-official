Add

The 'Add' method is commonly used in collections and dictionaries in C# to add new elements or key-value pairs to the collection. For lists, it appends an item to the end of the list, while for dictionaries, it adds a new key along with its associated value. This method is essential for dynamically managing collections as it allows the user to build and manipulate data structures during runtime.

using System;
using System.Collections.Generic;

class Program {
    static void Main() {
        List pythonExamples = new List();
        pythonExamples.Add("print('Hello World')");
        pythonExamples.Add("def greet(name):\n    print('Hello', name)");
        pythonExamples.Add("for i in range(5):\n    print(i)");

        foreach (var example in pythonExamples) {
            Console.WriteLine(example);
        }
    }
}
using System;
using System.Collections.Generic;

class Program {
    static void Main() {
        Dictionary codeSnippets = new Dictionary();
        codeSnippets.Add("example1", "print('Hello from example 1')");
        codeSnippets.Add("example2", "if x > 10:\n    print('x is greater than 10')");
        codeSnippets.Add("example3", "while True:\n    print('This will run forever')");

        foreach (var entry in codeSnippets) {
            Console.WriteLine(entry.Key + ": " + entry.Value);
        }
    }
}
using System;
using System.Collections.Generic;

class Program {
    static void Main() {
        List examples = new List();

        // Add a single Python code snippet
        examples.Add("def calculate_square(n):\n    return n * n");

        // Add multiple Python code snippets
        examples.Add("for j in range(3):\n    print(calculate_square(j))");
        examples.Add("def main():\n    calculate_square(5)");

        foreach (var snippet in examples) {
            Console.WriteLine(snippet);
        }
    }
}
