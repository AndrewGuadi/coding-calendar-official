acos

The 'acos' method in Java is a static method of the java.lang.Math class. It calculates the arc cosine of a given value, which is the angle (in radians) whose cosine is the specified value. The input value should be in the range of -1.0 to 1.0, inclusive. If the value is outside this range, the method will return NaN (Not a Number). The result can be further converted to degrees by using a conversion formula, if needed.

import java.lang.Math;

public class Main {
    public static void main(String[] args) {
        double value = 0.5;
        double angleInRadians = Math.acos(value);
        System.out.println("Arc Cosine of " + value + " is: " + angleInRadians);
    }
}
import java.lang.Math;

public class Main {
    public static void main(String[] args) {
        double value = -1.0;
        double angleInRadians = Math.acos(value);
        System.out.println("Arc Cosine of " + value + " is: " + angleInRadians);
    }
}
import java.lang.Math;

public class Main {
    public static void main(String[] args) {
        double value = 1.0;
        double angleInRadians = Math.acos(value);
        double angleInDegrees = Math.toDegrees(angleInRadians);
        System.out.println("Arc Cosine of " + value + " is: " + angleInRadians + " radians or " + angleInDegrees + " degrees.");
    }
}
