acos

The 'acos' method in Java is used to compute the arc cosine (inverse cosine) of a value. It returns the angle (in radians) whose cosine is the specified value. The value must be in the range [-1.0, 1.0]; otherwise, the method will return NaN (Not a Number). This method is a part of the java.lang.Math class.

import java.lang.Math;

public class AcosExample1 {
    public static void main(String[] args) {
        double value = 0.5;
        double result = Math.acos(value);
        System.out.println("The arc cosine of " + value + " is: " + result + " radians."); // The arc cosine of 0.5 is: 1.0471975511965979 radians.
    }
}
import java.lang.Math;

public class AcosExample2 {
    public static void main(String[] args) {
        double value = -1.0;
        double result = Math.acos(value);
        System.out.println("The arc cosine of " + value + " is: " + result + " radians."); // The arc cosine of -1.0 is: 3.141592653589793 radians.
    }
}
import java.lang.Math;

public class AcosExample3 {
    public static void main(String[] args) {
        double value = 2.0; // Out of range
        double result = Math.acos(value);
        System.out.println("The arc cosine of " + value + " is: " + result + ", which is NaN (Not a Number)."); // The arc cosine of 2.0 is: NaN (Not a Number).
    }
}
