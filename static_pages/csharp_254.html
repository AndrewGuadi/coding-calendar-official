Add

The 'Add' method in C# is commonly used to add an element to a collection, such as a list, dictionary, or set. This method typically takes a single parameter, which represents the element to be added. If the collection does not allow duplicate entries (like a set or dictionary keys), then attempting to add an existing element may throw an exception or result in a no-op, depending on the implementation. This method is crucial for dynamically managing data within collections during runtime.

class Program:
    def __init__(self):
        self.items = []

    def add_item(self, item):
        self.items.append(item)

    def show_items(self):
        for item in self.items:
            print(item)

# Usage
program = Program()
program.add_item('Python')
program.add_item('C#')
program.show_items()  # Output: Python, C#
class ShoppingCart:
    def __init__(self):
        self.cart = []

    def add_to_cart(self, product):
        self.cart.append(product)

    def view_cart(self):
        return self.cart

# Usage
shopping_cart = ShoppingCart()
shopping_cart.add_to_cart('Apple')
shopping_cart.add_to_cart('Banana')
print(shopping_cart.view_cart())  # Output: ['Apple', 'Banana']
class UniqueSet:
    def __init__(self):
        self.items = set()

    def add_unique(self, item):
        self.items.add(item)

    def get_items(self):
        return list(self.items)

# Usage
unique_set = UniqueSet()
unique_set.add_unique('Dog')
unique_set.add_unique('Cat')
unique_set.add_unique('Dog')  # No duplicates
print(unique_set.get_items())  # Output could be in any order: ['Dog', 'Cat']
