std::vector::push_back

The 'push_back' method in C++ is used to add a new element to the end of a vector. Vectors are dynamic arrays that can grow in size, and 'push_back' allows for seamless appending of elements without needing to manage the underlying storage manually. When an element is added, if the current allocation of memory for the vector is insufficient to store the new element, the vector automatically reallocates memory to accommodate the additional size. This method works for any data type that can be stored in a vector, including basic types and user-defined objects.

```cpp
#include 
#include 

int main() {
    std::vector languages;
    languages.push_back("Python");
    languages.push_back("C++");
    languages.push_back("Java");

    for (const auto& lang : languages) {
        std::cout << lang << " ";
    }
    return 0;
}
```

```cpp
    // Compute the sum of the elements
    int sum = 0;

    }
    std::cout << "Total Sum: " << sum << std::endl;
    return 0;
}
```

```cpp
int main() {
    std::vector temperatures;
    temperatures.push_back(98.6);
    temperatures.push_back(99.1);
    temperatures.push_back(97.5);

    // Output the average temperature
    double total = 0.0;
    for (double temp : temperatures) {
        total += temp;
    }
    double average = total / temperatures.size();
    std::cout << "Average Temperature: " << average << std::endl;
    return 0;
}
```
