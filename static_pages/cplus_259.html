std::vector::push_back

The `std::vector::push_back` method is used to add a new element to the end of a vector in C++. This operation increases the size of the vector by one and copies or moves the provided value into the added element. If the vector's current capacity is exceeded, it will automatically allocate more space to accommodate the new element. This method is essential for dynamic data handling as it allows flexibility in managing collections of objects or values without needing to declare a fixed size in advance.

```cpp
#include 
#include 

int main() {
    std::vector myVector;
    myVector.push_back("Hello");
    myVector.push_back("World");

    for (const auto& word : myVector) {
        std::cout << word << " ";
    }
    return 0;
}
```

#include

int main() {
    std::vector numbers;
    numbers.push_back(1);
    numbers.push_back(2);
    numbers.push_back(3);

    int sum = 0;
    for (const int& number : numbers) {
        sum += number;
    }
    std::cout << "Sum: " << sum;
    return 0;
}

int main() {
    std::vector languages;
    languages.push_back("Python");
    languages.push_back("C++");
    languages.push_back("JavaScript");

    std::cout << "Languages: ";
    for (std::size_t i = 0; i < languages.size(); ++i) {
        std::cout << languages[i];
        if (i < languages.size() - 1) std::cout << ", ";
    }
    std::cout << std::endl;
    return 0;
}
