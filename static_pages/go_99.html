cap

In Go, the 'cap' function is used to determine the capacity of a slice, which is the number of elements the slice can hold before needing to allocate more memory. The capacity can be equal to or greater than the length of the slice, and it helps in optimizing memory usage, especially when dealing with dynamic data structures. It is important to note that 'cap' returns the capacity of the underlying array that the slice references, not the number of elements currently in the slice (which is given by 'len').

```go
package main
import (
	"fmt"
)

func main() {
	// Create a slice with an initial capacity of 3
	slice := make([]int, 0, 3)
	fmt.Println("Length:", len(slice))  // Output: Length: 0
	fmt.Println("Capacity:", cap(slice)) // Output: Capacity: 3

	// Append elements to the slice
	slice = append(slice, 1, 2, 3)
	fmt.Println("Length after appending 3 elements:", len(slice))  // Output: Length: 3
	fmt.Println("Capacity after appending 3 elements:", cap(slice)) // Output: Capacity: 3

	// Append another element to the slice
	slice = append(slice, 4)
	fmt.Println("Length after appending 4:", len(slice))  // Output: Length: 4
	fmt.Println("Capacity after appending 4:", cap(slice)) // Output: Capacity: 6 (increased)
}
```
```go
package main
import (
	"fmt"
)

func main() {
	// Create a slice with an initial capacity of 5
	mySlice := make([]string, 0, 5)
	fmt.Println("Length:", len(mySlice))  // Output: Length: 0
	fmt.Println("Capacity:", cap(mySlice)) // Output: Capacity: 5

	// Adding elements
	mySlice = append(mySlice, "apple", "banana")
	fmt.Println("Length after adding fruits:", len(mySlice))  // Output: Length: 2
	fmt.Println("Capacity after adding fruits:", cap(mySlice)) // Output: Capacity: 5
}
```

```go
package main
import (
	"fmt"
)

func main() {
	// Create a zero-length slice with a capacity of 10
	arraySlice := make([]float64, 0, 10)
	fmt.Println("Initial Length:", len(arraySlice))  // Output: Initial Length: 0
	fmt.Println("Initial Capacity:", cap(arraySlice)) // Output: Initial Capacity: 10

	// Adding elements
	for i := 0; i < 10; i++ {
		arraySlice = append(arraySlice, float64(i)*1.1)
	}
	fmt.Println("Length after loop:", len(arraySlice))   // Output: Length: 10
	fmt.Println("Capacity after loop:", cap(arraySlice))  // Output: Capacity: 10

	// Appending one more element forcefully increases capacity
	arraySlice = append(arraySlice, 11.1)
	fmt.Println("Length after adding one more element:", len(arraySlice))   // Output: Length: 11
	fmt.Println("Capacity after adding one more element:", cap(arraySlice))  // Output: Capacity: 20 (increased)
}
```
