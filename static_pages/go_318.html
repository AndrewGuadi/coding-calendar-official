close

The 'close' method in Go is typically used to close a resource such as a file, network connection, or channel, signaling that the resource is no longer needed. Closing resources is an important practice to ensure that memory and other system resources are freed up properly and to prevent memory leaks or resource exhaustion. When closing a file, for instance, it ensures that all buffered data is written and that the file descriptor is released. The close method is an essential part of resource management in Go.

import os

func closeFile(file *os.File) {
    defer file.Close() // Ensure that we close the file when we're done
    // Do something with the file
}

func main() {
    file, err := os.Open("example.txt")
    if err != nil {
        panic(err)
    }
    closeFile(file)
}
package main

import (
    "fmt"
    "net"
)

func closeConnection(conn net.Conn) {
    defer func() {
        if err := conn.Close(); err != nil {
            fmt.Println("Error closing connection:", err)
        }
    }()
    // Do something with the connection
}

func main() {
    conn, err := net.Dial("tcp", "localhost:8080")
    if err != nil {
        panic(err)
    }
    closeConnection(conn)
}
package main

import "fmt"

func closeChannel(ch chan int) {
    defer close(ch) // Defer the closure of channel until function completes
    // Do something with the channel
}

func main() {
    ch := make(chan int)
    closeChannel(ch)
    // Cannot send to closed channel: runtime error
    // ch <- 1 // Uncommenting this will cause a panic
    fmt.Println("Channel closed")
}
