close

In Go, the 'close' method is used to close a channel. Once a channel is closed, no more values can be sent to it, but it can still be received. Closing a channel notifies all the receivers that no more values will be sent. It is important to close channels to prevent resource leaks.

def worker(ch):
    for msg in ch:
        print(f'Received: {msg}')

ch = []
for i in range(5):
    ch.append(i)

# Close the channel
ch.close()

worker(ch)
def producer(ch):
    for i in range(5):
        ch.append(i)
    ch.close()  # Close the channel when done

ch = []
producer(ch)

for msg in ch:
    print(f'Got message: {msg}')
def main():
    ch = []
    def sender(ch):
        ch.extend(range(3))
        ch.close()
    sender(ch)

    for msg in ch:
        print(f'Sending: {msg}')

main()
