abs

The 'abs' method in Java is used to return the absolute value of a given number. It can be applied to different numeric types, including integers, floating-point numbers, and long integers. The absolute value of a number is its distance from zero, without considering the direction. For instance, the absolute value of both -5 and 5 is 5. This method is part of the java.lang.Math class and can be used conveniently in any Java application.

import java.util.Scanner;

public class AbsoluteExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int number = scanner.nextInt();
        int absoluteValue = Math.abs(number);
        System.out.println("The absolute value is: " + absoluteValue);
    }
}
public class FloatAbsExample {
    public static void main(String[] args) {
        double negativeFloat = -45.67;
        double absoluteFloat = Math.abs(negativeFloat);
        System.out.println("The absolute value of " + negativeFloat + " is: " + absoluteFloat);
    }
}
public class LongAbsExample {
    public static void main(String[] args) {
        long negativeLong = -123456789L;
        long absoluteLong = Math.abs(negativeLong);
        System.out.println("The absolute value of " + negativeLong + " is: " + absoluteLong);
    }
}
