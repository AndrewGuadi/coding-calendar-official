close

In Go, the 'close' function is used to close a channel, which signals that no more values will be sent on that channel. Once a channel is closed, attempts to send values on it will cause a panic, though it is safe to close a channel multiple times. After closure, a range loop on a closed channel will continue to read any values that were sent before the channel was closed. Closing a channel is particularly important to prevent goroutines from hanging and waiting for values that will never arrive.

def close_channel_example():
    import threading
    import time
    from queue import Queue

    channel = Queue()

    def producer():
        for i in range(5):
            channel.put(i)
            print(f"Produced: {i}")
            time.sleep(0.5)
        channel.put(None)  # Signal that we're done producing

    def consumer():
        while True:
            item = channel.get()
            if item is None:  # Stop if None is received
                print("Consumer: Received stop signal")
                break
            print(f"Consumed: {item}")

    producer_thread = threading.Thread(target=producer)
    consumer_thread = threading.Thread(target=consumer)
    producer_thread.start()
    consumer_thread.start()
    producer_thread.join()
    consumer_thread.join()

def close_channel_example_2():
    import threading
    import time
    from queue import Queue

    channel = Queue(maxsize=2)  # Create a bounded channel

    def producer():
        for i in range(5):
            channel.put(i)
            print(f"Produced: {i}")
            time.sleep(0.5)
        channel.put(None)  # Signal that we're done producing

    def consumer():
        while True:
            item = channel.get()
            if item is None:
                print("Consumer: Shutting down...")
                channel.task_done()  # Signal that we received the None
                break
            print(f"Consumed: {item}")
            channel.task_done()

    producer_thread = threading.Thread(target=producer)
    consumer_thread = threading.Thread(target=consumer)
    producer_thread.start()
    consumer_thread.start()
    producer_thread.join()
    consumer_thread.join()

def close_channel_example_3():
    import threading
    import time
    from queue import Queue

    channel = Queue()

    def producer():
        for i in range(3):
            channel.put(i)
            print(f"Produced: {i}")
            time.sleep(1)
        channel.put(None)  # Signal completion

    def consumer():
        while True:
            item = channel.get()
            if item is None:
                print("No more items to consume. Closing...")
                break
            print(f"Consumed: {item}")

    producer_thread = threading.Thread(target=producer)
    consumer_thread = threading.Thread(target=consumer)
    producer_thread.start()
    consumer_thread.start()
    producer_thread.join()
    channel.put(None)  # Close the channel
    consumer_thread.join()
