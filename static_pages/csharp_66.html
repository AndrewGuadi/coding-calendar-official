Add

The 'Add' method is commonly used in C# to add elements to collections such as lists, dictionaries, or sets. It allows the user to insert a new element into the collection, expanding its size. This method is essential for managing dynamic data and is widely used in various applications where collections are modified frequently.

class Program:
    def __init__(self):
        self.my_list = []

    def add_to_list(self, item):
        self.my_list.append(item)
        return self.my_list

    def main():
        program = Program()
        print(program.add_to_list("Hello, World!"))  # Output: ["Hello, World!"]
        print(program.add_to_list("Python is great!"))  # Output: ["Hello, World!", "Python is great!"]

    if __name__ == '__main__':
        main()
class ShoppingCart:
    def __init__(self):
        self.items = []

    def add_item(self, item):
        self.items.append(item)
        return self.items

    def main():
        cart = ShoppingCart()
        print(cart.add_item("Apples"))  # Output: ["Apples"]
        print(cart.add_item("Bananas"))  # Output: ["Apples", "Bananas"]

    if __name__ == '__main__':
        main()
class ContactList:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, phone):
        self.contacts[name] = phone
        return self.contacts

    def main():
        contacts = ContactList()
        print(contacts.add_contact("Alice", "123-456-7890"))  # Output: {"Alice": "123-456-7890"}
        print(contacts.add_contact("Bob", "987-654-3210"))  # Output: {"Alice": "123-456-7890", "Bob": "987-654-3210"}

    if __name__ == '__main__':
        main()
