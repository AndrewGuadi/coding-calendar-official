at

The `at` method in C++ is a member function of several Standard Library (STL) containers like `std::vector`, `std::array`, and `std::map`. It is used to access the element at a specified position with bounds checking. If the position provided is out-of-bounds, the method throws an `std::out_of_range` exception, making it safer compared to the subscript operator `[]` which does not perform bounds checking. This method provides a way to safely access elements, ensuring that your program can handle out-of-bound errors gracefully.

int main() {
    // Example with std::vector
    std::vector vec = {1, 2, 3, 4, 5};
    try {
        std::cout << "Element at index 2 in vector: " << vec.at(2) << std::endl;
        std::cout << "Element at index 5 in vector: " << vec.at(5) << std::endl;  // This will throw an exception
    } catch (const std::out_of_range& e) {
        std::cerr << "Out of range error: " << e.what() << std::endl;
    }

    // Example with std::array
    std::array arr = {10, 20, 30, 40, 50};
    try {
        std::cout << "Element at index 3 in array: " << arr.at(3) << std::endl;
        std::cout << "Element at index 10 in array: " << arr.at(10) << std::endl;  // This will throw an exception
    } catch (const std::out_of_range& e) {
        std::cerr << "Out of range error: " << e.what() << std::endl;
    }

    // Example with std::map
    std::map ages = { {"Alice", 30}, {"Bob", 25}, {"Charlie", 35} };
    try {
        std::cout << "Age of Bob: " << ages.at("Bob") << std::endl;
        std::cout << "Age of Daniel: " << ages.at("Daniel") << std::endl;  // This will throw an exception
    } catch (const std::out_of_range& e) {
        std::cerr << "Out of range error: " << e.what() << std::endl;
    }

    return 0;
}
