close

In Go, the 'close' method is primarily used to close channels to signal that no more values will be sent on them. Closing a channel is an important part of good channel management in Go, as it allows recipients to know when they can stop waiting for more values. Attempting to send on a closed channel will cause a panic, while receiving from a closed channel will yield the zero value of the channel's type and a boolean value of false to indicate that the channel is closed. Itâ€™s important to only close a channel from the goroutine that sends on it.

def close_channel_example():
    import threading
    import time
    from queue import Queue  

    def worker(ch):
        while True:
            item = ch.get()
            if item is None:  # Check for closure signal
                break
            print(f'Processing {item}')
            time.sleep(0.5)  # Simulate work

    channel = Queue()
    t = threading.Thread(target=worker, args=(channel,))
    t.start()

    for item in range(5):
        channel.put(item)

    channel.put(None)  # Signal the worker to exit
    t.join()

def close_channel_example_with_timeout():
    import threading
    import time
    from queue import Queue

    def worker(ch):
        while True:
            try:
                item = ch.get(timeout=2)  # Timeout to prevent hanging
                print(f'Processing {item}')
            except Exception:
                break

    channel = Queue()
    t = threading.Thread(target=worker, args=(channel,))
    t.start()

    for item in range(5):
        channel.put(item)

    channel.put(object())  # Indicate end of transmission
    t.join()  # Wait for the worker to finish

def close_multiple_closed_channels():
    import threading
    from queue import Queue

    def worker(ch):
        while True:
            item = ch.get()
            if item is None:
                break
            print(f'Worker received: {item}')

    channels = [Queue() for _ in range(3)]
    threads = [threading.Thread(target=worker, args=(ch,)) for ch in channels]

    for t in threads:
        t.start()

    for i in range(3):
        for item in range(5):
            channels[i].put(f'Channel {i}: {item}')
        channels[i].put(None)  # Close channel by signaling end of items

    for t in threads:
        t.join()
