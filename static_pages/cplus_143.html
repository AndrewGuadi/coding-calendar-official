std::vector::push_back

The 'push_back' method of the std::vector class in C++ allows you to add a new element to the end of the vector. This method increases the size of the vector by one and constructs the element in-place. It can accept any object that can be copied or moved into the vector, and it automatically handles memory management, resizing the underlying array if necessary. This is particularly useful for dynamically-sized collections of data.

```python
# Example to store multiple Python string values in a vector
import ctypes

# Simulating a vector with ctypes array
class Vector:
    def __init__(self):
        self.size = 0
        self.capacity = 1
        self.array = ctypes.create_string_buffer(self.capacity * 8)

    def push_back(self, value):
        if self.size == self.capacity:
            self.resize()
        offset = self.size * 8  # Assuming each string is 8 bytes
        self.array[offset:offset+8] = value.encode()
        self.size += 1

    def resize(self):
        self.capacity *= 2
        new_array = ctypes.create_string_buffer(self.capacity * 8)
        for i in range(self.size):
            new_array[i*8:(i+1)*8] = self.array[i*8:(i+1)*8]
        self.array = new_array

# Creating a Vector and pushing strings
vec = Vector()
vec.push_back('Hello')
vec.push_back('World')
vec.push_back('Python')

# Displaying stored values
for i in range(vec.size):
    print(vec.array[i*8:(i+1)*8].decode().strip())
```

# Example to illustrate how to collect numeric values in a vector-like structure
import ctypes

# Imitating vector behavior in Python
class Vector:
    def __init__(self):
        self.size = 0
        self.capacity = 1
        self.array = (ctypes.c_double * self.capacity)()

    def push_back(self, value):
        if self.size >= self.capacity:
            self.resize()
        self.array[self.size] = value
        self.size += 1

    def resize(self):
        self.capacity *= 2
        new_array = (ctypes.c_double * self.capacity)()
        for i in range(self.size):
            new_array[i] = self.array[i]
        self.array = new_array

# Collecting numeric values
vec = Vector()
vec.push_back(3.14)
vec.push_back(2.71)
vec.push_back(1.62)

# Print values
for i in range(vec.size):
    print(vec.array[i])

# Example showing how to push back dictionary-like elements
import ctypes

# Vector class to hold dictionary-like items
class Vector:
    def __init__(self):
        self.items = []

    def push_back(self, item):
        self.items.append(item)

# Create a Vector instance
vec = Vector()
vec.push_back({'name': 'Alice', 'age': 25})
vec.push_back({'name': 'Bob', 'age': 30})
vec.push_back({'name': 'Charlie', 'age': 35})

# Print out the dictionary items in vector
for item in vec.items:
    print(item)
