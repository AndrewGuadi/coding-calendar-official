abs

The 'abs' method in Java is used to compute the absolute value of a number. It can be applied to different data types including integers, floats, and doubles. The absolute value of a number is defined as its non-negative value; that is, if the value is positive, it remains the same, and if it's negative, it becomes positive. This method is particularly useful in mathematical computations where the magnitude of a number is needed without regard to its sign.

import java.util.Scanner;

public class AbsoluteValueExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int number = scanner.nextInt();
        System.out.println("The absolute value of " + number + " is: " + Math.abs(number));
    }
}

# Python equivalent:
number = -5
absolute_value = abs(number)
print(f'The absolute value of {number} is: {absolute_value}')
import java.util.Scanner;

public class AbsoluteDoubleValueExample {

        System.out.print("Enter a floating-point number: ");

}

# Python equivalent:
number = -12.5
absolute_value = abs(number)
print(f'The absolute value of {number} is: {absolute_value}')
import java.util.Scanner;

public class AbsoluteArrayValueExample {
    public static void main(String[] args) {
        int[] numbers = {-4, 7, -2, 3};
        System.out.println("Absolute values of the array:");
        for (int number : numbers) {
            System.out.println(Math.abs(number));
        }
    }
}

# Python equivalent:
numbers = [-4, 7, -2, 3]
absolute_values = [abs(number) for number in numbers]
print(f'Absolute values of the array: {absolute_values}')
