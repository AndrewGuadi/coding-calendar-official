close

In Go, the 'close' method is primarily used to close channels. Closing a channel prevents further sends on the channel and is typically used to signal that no more values will be sent on it. It is important to note that closing a channel is a signal for receivers to stop waiting for new values. The 'close' function can only be called once on a channel and closing a channel that is already closed will cause a panic. Itâ€™s a best practice to use the 'close' method in conjunction with a 'for range' loop to safely receive all values until the channel is closed.

def channel_example():
    import time
    import threading
    
    def producer(ch):
        for i in range(5):
            ch.append(i)
            time.sleep(0.5)
        # Close the channel
        print('Closing channel')
        ch.append('close')

    def consumer(ch):
        while True:
            item = ch.pop(0)
            if item == 'close':
                break
            print('Received:', item)

    channel = []
    t1 = threading.Thread(target=producer, args=(channel,))
    t2 = threading.Thread(target=consumer, args=(channel,))
    t1.start()
    t2.start()
    t1.join()
    t2.join()
def close_signal_example():
    import threading
    import time
    
    def worker(events):
        while not events['exit']:
            print('Working...')
            time.sleep(1)
        print('Worker received exit signal.')

    events = {'exit': False}
    worker_thread = threading.Thread(target=worker, args=(events,))
    worker_thread.start()
    time.sleep(5)  # let the worker run for a while
    print('Sending close signal')
    events['exit'] = True  # Close the exit signal
    worker_thread.join()
    print('Worker thread has closed.')
def task_manager():
    import threading
    import queue
    import time
    
    def worker(task_queue):
        while True:
            task = task_queue.get()
            if task is None:
                break
            print(f'Processing task {task}')
            time.sleep(1)
            task_queue.task_done()

    task_queue = queue.Queue()
    threads = []
    for _ in range(3):  # Start 3 worker threads
        t = threading.Thread(target=worker, args=(task_queue,))
        t.start()
        threads.append(t)

    for i in range(10):  # Add tasks
        task_queue.put(i)

    task_queue.join()  # Wait for all tasks to be processed
    print('All tasks queued.')

    # Close the workers
    for _ in threads:
        task_queue.put(None)  # Signal all workers to exit
    for t in threads:
        t.join()  # Wait for all threads to finish
    print('All workers have closed.')
