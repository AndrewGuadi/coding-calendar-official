close

The 'close' method is used to close a resource that was previously opened, such as a file, network connection, or database connection. Closing a resource ensures that all resources associated with it are released, preventing memory leaks and ensuring that any unsaved data is properly flushed and saved. This method should always be called once the resource is no longer needed.

import json

# Example of using close with a file
file_path = 'example.json'
with open(file_path, 'w') as file:
    json.dump({'key': 'value'}, file)

# Closing the file explicitly (not necessary when using 'with')
file.close()
import socket

# Example of using close with a network socket
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
sock.connect(('localhost', 8080))

# Sending data
sock.sendall(b'Hello, World!')

# Closing the socket
sock.close()
import sqlite3

# Example of using close with a database connection
db_connection = sqlite3.connect('example.db')
cursor = db_connection.cursor()

# Execute a query
cursor.execute('CREATE TABLE IF NOT EXISTS example (id INTEGER PRIMARY KEY, name TEXT)')

# Close the cursor and connection
cursor.close()
db_connection.close()
