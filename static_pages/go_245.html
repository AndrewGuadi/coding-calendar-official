close

In Go, the 'close' method is used to close a channel, which prevents any further values from being sent on that channel. Closing a channel will not affect any ongoing reads from that channel, but any attempts to send data on a closed channel will cause a panic. It's good practice to close a channel when it is no longer needed, typically from the goroutine that sends data to the channel. To reliably ascertain whether a channel is closed, a two-value assignment can be used to check the status during read operations.

```python
import threading
import time

def worker(ch):
    for i in range(5):
        ch.append(i)
        time.sleep(1)
    # closing the channel/ch list by sending a termination signal
    ch.append(None)  # signaling completion

# Channel as a list
channel = []
thread = threading.Thread(target=worker, args=(channel,))
thread.start()

while True:
    value = channel.pop(0) if channel else None
    if value is None:
        print('Channel closed, exiting...')
        break  # exiting when channel is closed
    else:
        print(f'Received: {value}')

```python
import threading
import time

def producer(ch):
    for i in range(5):
        ch.append(i)
        time.sleep(0.5)
    # signaling that no more data will be sent
    ch.append(None)  # Closing the channel

def consumer(ch):
    while True:
        if ch:
            value = ch.pop(0)
            if value is None:
                print('Consumer: Channel closed.')
                break
            print(f'Consumer received: {value}')
        else:
            time.sleep(0.1)  # Avoid busy-waiting

channel = []
thread_producer = threading.Thread(target=producer, args=(channel,))
thread_consumer = threading.Thread(target=consumer, args=(channel,))
thread_producer.start()
thread_consumer.start()
thread_producer.join()
thread_consumer.join()

```python
import threading
import time

def number_generator(ch):
    for i in range(10):
        ch.append(i)
        time.sleep(0.5)
    ch.append(None)  # signal that we're done

def number_consumer(ch):
    while True:
        if ch:
            num = ch.pop(0)  # read from channel
            if num is None:
                print('Consumer: Channel closed, stopping.')
                break
            print(f'Consumed number: {num}')
        else:
            time.sleep(0.1)  # wait for new values

channel = []
producer_thread = threading.Thread(target=number_generator, args=(channel,))
consumer_thread = threading.Thread(target=number_consumer, args=(channel,))
producer_thread.start()
consumer_thread.start()
producer_thread.join()
consumer_thread.join()
