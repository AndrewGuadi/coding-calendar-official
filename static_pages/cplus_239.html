std::vector::push_back

The `push_back` method is a member function of the `std::vector` class in C++ that allows you to add an element to the end of a vector. It resizes the vector if necessary and inserts the new element at the back. This method is useful for dynamically growing the vector and managing collections of data that may change in size during program execution.

# Example 1: Adding elements to a vector
#include 
#include 

int main() {
    std::vector languages;
    
    // Adding elements to the vector
    languages.push_back("Python");
    languages.push_back("Java");
    languages.push_back("C++");
    
    // Displaying the elements in the vector
    for(const auto& language : languages) {
        std::cout << language << std::endl;
    }
    return 0;
}

int main() {
    std::vector fruits;
    std::string fruitList[] = {"Apple", "Banana", "Cherry"};
    
    // Adding elements using a loop
    for (const auto& fruit : fruitList) {
        fruits.push_back(fruit);
    }
    
    // Displaying the fruits
    for (const auto& fruit : fruits) {
        std::cout << fruit << std::endl;
    }
    return 0;
}

    std::vector numbers;
