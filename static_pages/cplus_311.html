std::vector::push_back

The std::vector::push_back method adds a new element to the end of the vector, increasing the container size by one. This method is used when you want to add an element dynamically without worrying about the current size of the vector. The new element is a copy of the value passed to push_back. If the vector's current capacity is reached, it automatically resizes itself to accommodate the new element.

```python
# Example of using std::vector::push_back in a C++ vector

#include 
#include 

int main() {
    std::vector languages;
    languages.push_back("Python");
    languages.push_back("C++");
    languages.push_back("JavaScript");

    // Display the contents of the vector
    for (const auto& lang : languages) {
        std::cout << lang << std::endl;
    }
    return 0;
}
```
```python
# Example of using std::vector::push_back to store integers in a vector

#include 
#include 

int main() {
    std::vector numbers;
    for (int i = 1; i <= 5; ++i) {
        numbers.push_back(i);
    }

    // Display the contents of the vector
    for (const auto& num : numbers) {
        std::cout << num << " ";
    }
    return 0;
}
```

int main() {
    std::vector stack;

    // Push elements onto the stack
    stack.push_back("first");
    stack.push_back("second");
    stack.push_back("third");

    // Pop elements from the stack (LIFO)
    while (!stack.empty()) {
        std::cout << stack.back() << std::endl;  // Access the last element
        stack.pop_back();  // Remove the last element
    }
    return 0;
}
