abs

The `abs` method in Java is used to compute the absolute value of a given number. This method is available in the Math class and can work with different numeric types such as integers, floating-point numbers (float and double), and long integers. The absolute value of a number is its distance from zero on the number line, regardless of direction. For example, the absolute value of -5 is 5, and the absolute value of 5 is also 5. This method helps in scenarios where the magnitude of a number is important, but its sign is not.

import java.lang.Math;

public class AbsoluteValueExample {
    public static void main(String[] args) {
        int number1 = -10;
        int number2 = 10;
        float number3 = -10.5f;
        double number4 = 10.5;
        
        // Using abs method to find absolute values
        System.out.println("Absolute value of " + number1 + " is: " + Math.abs(number1));  // Outputs 10
        System.out.println("Absolute value of " + number2 + " is: " + Math.abs(number2));  // Outputs 10
        System.out.println("Absolute value of " + number3 + " is: " + Math.abs(number3));  // Outputs 10.5
        System.out.println("Absolute value of " + number4 + " is: " + Math.abs(number4));  // Outputs 10.5
    }
}
import java.lang.Math;

public class TemperatureConverter {
    public static void main(String[] args) {
        float celsius = -15.0f;
        float fahrenheit = convertCelsiusToFahrenheit(celsius);
        System.out.println("The absolute temperature in Fahrenheit is: " + Math.abs(fahrenheit));
    }
    
    public static float convertCelsiusToFahrenheit(float celsius) {
        return (celsius * 9/5) + 32;
    }
}
import java.util.Scanner;
import java.lang.Math;

public class DistanceCalculator {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the x-coordinate of the first point: ");
        double x1 = scanner.nextDouble();
        System.out.print("Enter the y-coordinate of the first point: ");
        double y1 = scanner.nextDouble();
        System.out.print("Enter the x-coordinate of the second point: ");
        double x2 = scanner.nextDouble();
        System.out.print("Enter the y-coordinate of the second point: ");
        double y2 = scanner.nextDouble();
        double distance = Math.sqrt(Math.pow(Math.abs(x2 - x1), 2) + Math.pow(Math.abs(y2 - y1), 2));
        System.out.println("The distance between the two points is: " + distance);
    }
}
