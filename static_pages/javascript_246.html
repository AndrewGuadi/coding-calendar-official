charAt

The `charAt` method in JavaScript returns the character at a specified index (position) in a string. The index is zero-based, meaning that the first character of the string is at index 0, the second character is at index 1, and so forth. If the specified index is out of range (i.e., negative or greater than the string's length minus one), an empty string is returned.

let str = 'Hello, World!';
console.log(str.charAt(0)); // Outputs: 'H'
console.log(str.charAt(4)); // Outputs: 'o'
console.log(str.charAt(13)); // Outputs: '' (empty string)
console.log(str.charAt(-1)); // Outputs: '' (empty string)
let greeting = 'Good morning!';
let index = 4;
console.log(greeting.charAt(index)); // Outputs: 'm', as it is the character at index 4
let alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
for (let i = 0; i < alphabet.length; i++) {
    console.log('Character at index ' + i + ': ' + alphabet.charAt(i));
}
// Outputs each character of the alphabet.
