Add

The 'Add' method is used to add an element to a collection, such as a list or a dictionary. In C#, this method is commonly associated with collection classes like List or Dictionary. When using the 'Add' method, the item is typically appended to the end of the collection in the case of lists, or associated with a specified key in the case of dictionaries. This allows for dynamic expansion of collections during the runtime of a program.

class Program:
    def __init__(self):
        self.my_list = []

    def add_to_list(self, item):
        self.my_list.append(item)

    def display_list(self):
        print(self.my_list)

# Usage of Add Example
program = Program()
program.add_to_list('Python')
program.add_to_list('C#')
program.display_list()  # Output: ['Python', 'C#']

class Program:
    def __init__(self):
        self.my_dict = {}

    def add_to_dict(self, key, value):
        self.my_dict[key] = value

    def display_dict(self):
        print(self.my_dict)

# Usage of Add Example
program = Program()
program.add_to_dict('name', 'Alice')
program.add_to_dict('language', 'Python')
program.display_dict()  # Output: {'name': 'Alice', 'language': 'Python'}

class Program:
    def __init__(self):
        self.my_items = []

    def add_multiple_items(self, items):
        for item in items:
            self.my_items.append(item)

    def display_items(self):
        print(self.my_items)

# Usage of Add Multiple Example
program = Program()
program.add_multiple_items(['Java', 'C++', 'JavaScript'])
program.display_items()  # Output: ['Java', 'C++', 'JavaScript']
