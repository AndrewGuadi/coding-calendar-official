close

The 'close' method in Go is typically used to close resources such as files, network connections, or channels. Closing a resource frees up any system resources associated with it and ensures that no further operations are performed on it. In the context of files, closing ensures that all buffered operations are flushed and that the file descriptors are properly released. Itâ€™s important to always check for errors during closing operations to avoid unexpected behavior.

import os

# Function to write and read a temporary file

def write_and_read_file():
    # Creating a temporary file
    file, err = os.Create('tempfile.txt')
    if err != nil:
        print('Error creating file:', err)
        return
    
    # Writing to the file
    _, err = file.write(b'Hello, world!')
    if err != nil:
        print('Error writing to file:', err)
        return
    
    # Closing the file
    err = file.close()
    if err != nil:
        print('Error closing file:', err)
        return
    
    # Reading from the file
    file, err = os.Open('tempfile.txt')
    if err != nil:
        print('Error opening file:', err)
        return
    
    data = file.read()
    print('File contents:', data)
    file.close()

# Call the function
write_and_read_file()
import socket

# Function to create a TCP connection

def create_tcp_connection():
    # Creating a TCP socket
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    try:
        sock.connect(('localhost', 8080))
        print('Connected to server.')
        # Send some data
        sock.sendall(b'Hello, server!')
    except Exception as e:
        print('Error:', e)
    finally:
        # Always close the socket
        sock.close()
        print('Socket closed.')

# Call the function
create_tcp_connection()
import threading

# Function to implement a producer-consumer model using a channel

def producer_consumer():
    channel = []
    lock = threading.Lock()
    
    def producer():
        for i in range(5):
            lock.acquire()
            channel.append(i)
            print('Produced:', i)
            lock.release()

    def consumer():
        for _ in range(5):
            lock.acquire()
            if channel:
                item = channel.pop(0)
                print('Consumed:', item)
            lock.release()

    # Starting threads
    p = threading.Thread(target=producer)
    c = threading.Thread(target=consumer)
    p.start()
    c.start()
    p.join()
    c.join()

# Call the function
producer_consumer()
