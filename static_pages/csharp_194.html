Add

The `Add` method in C# is generally used to add an element to a collection, such as a list, a dictionary, or a set. It typically modifies the collection by inserting the new element provided as an argument. Usage of `Add` method can vary depending on the type of collection being used. For example, in a List, it adds an item to the end of the list, while in a Dictionary it adds a key-value pair. It is important to ensure that the collection allows duplicates (for collections like List) or does not allow them (for Dictionary) based on the type of data structure being used.

using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        List pythonCodes = new List();

        // Adding Python code to the List
        pythonCodes.Add("print('Hello, World!')");
        pythonCodes.Add("for i in range(5): print(i)");
        pythonCodes.Add("def greet(name): return f'Hello {name}'");

        // Displaying the list
        foreach(var code in pythonCodes)
        {
            Console.WriteLine(code);
        }
    }
} 
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Dictionary codesDictionary = new Dictionary();

        // Adding Python code snippets with names as keys
        codesDictionary.Add("HelloWorld", "print('Hello, World!')");
        codesDictionary.Add("LoopExample", "for i in range(5): print(i)");
        codesDictionary.Add("FunctionExample", "def greet(name): return f'Hello {name}'");

        // Displaying the dictionary
        foreach(var entry in codesDictionary)
        {
            Console.WriteLine(entry.Key + ": " + entry.Value);
        }
    }
}
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        HashSet pythonSet = new HashSet();

        // Adding unique Python code snippets
        pythonSet.Add("print('Hello, World!')");
        pythonSet.Add("for i in range(5): print(i)");
        pythonSet.Add("def greet(name): return f'Hello {name}'");
        pythonSet.Add("def greet(name): return f'Hello {name}'"); // Duplicate, won't be added

        // Displaying the set
        foreach(var code in pythonSet)
        {
            Console.WriteLine(code);
        }
    }
}
