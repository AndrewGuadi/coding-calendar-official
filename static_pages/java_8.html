abs

The 'abs' method in Java is a static method provided by the Math class used to return the absolute value of a given number. It can take different data types as input such as int, long, float, and double. The absolute value of a number is its non-negative value regardless of its sign. For example, the absolute value of -5 is 5, and the absolute value of 5 is also 5. This method is useful in scenarios where the negative value needs to be converted to a positive one for calculations.

import java.util.Scanner; public class AbsoluteValueExample { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); System.out.print("Enter an integer: "); int number = scanner.nextInt(); int absoluteValue = Math.abs(number); System.out.println("The absolute value of " + number + " is: " + absoluteValue); scanner.close(); } }
public class DoubleAbsExample { public static void main(String[] args) { double negativeNumber = -99.75; double absValue = Math.abs(negativeNumber); System.out.println("The absolute value of " + negativeNumber + " is: " + absValue); } }
public class FloatAbsExample { public static void main(String[] args) { float negativeFloat = -56.23f; float absValue = Math.abs(negativeFloat); System.out.println("The absolute value of " + negativeFloat + " is: " + absValue); } }
