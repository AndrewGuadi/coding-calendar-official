acos

The 'acos' method in Java is used to compute the arc cosine (inverse cosine) of a given value. The result is returned in radians, which is a number between 0 and Ï€ (0 to 3.14159). The input value must be in the range of -1.0 to 1.0; otherwise, the method will throw an IllegalArgumentException. This method is found in the Math class, which provides mathematical functions. Typically, you would use acos when working with angles or trigonometric calculations, particularly in graphics, physics simulations, or game development.

import java.util.Scanner;

public class ArcCosExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a value between -1 and 1:");
        double value = scanner.nextDouble();
        if (value < -1 || value > 1) {
            System.out.println("Value must be within the range of -1 to 1.");
            return;
        }
        double result = Math.acos(value);
        System.out.println("The arc cosine of " + value + " is: " + result + " radians.");
    }
}
import java.lang.Math;

public class CircleExample {
    public static void main(String[] args) {
        double adjacent = 4.0;
        double hypotenuse = 5.0;
        double cosine = adjacent / hypotenuse; // Calculate cosine
        double angleInRadians = Math.acos(cosine);
        System.out.println("Angle in radians: " + angleInRadians);
    }
}
import java.util.ArrayList;

public class ArrayListExample {
    public static void main(String[] args) {
        ArrayList values = new ArrayList<>();
        values.add(-1.0);
        values.add(0.0);
        values.add(0.5);
        values.add(1.0);
        for (double value : values) {
            double angle = Math.acos(value);
            System.out.println("The arc cosine of " + value + " is: " + angle + " radians.");
        }
    }
}
