close

In Go, the 'close' method is used to close a channel, which signals that no more values will be sent on that channel. After a channel is closed, it is considered unsafe to send values on it; however, values can still be received from it until all of its values have been consumed. Attempting to send a value on a closed channel will cause a panic. Closing a channel can help to manage the lifecycle of goroutines that might be waiting for values on that channel, allowing them to terminate gracefully when no more values are forthcoming.

    time.sleep(1)  # Allow some time for the worker to finish

def go_close_example_2():
    import threading
    import queue
    import time

    def worker(ch):
        for val in iter(ch.get, None):
            print(f'Consumed: {val}')
        print('Worker exiting.')

    ch = queue.Queue()
    t = threading.Thread(target=worker, args=(ch,))
    t.start()

    for i in range(5):
        ch.put(i)
        print(f'Produced: {i}')
        time.sleep(0.5)

    ch.put(None)  # Close the channel by sending None
    t.join()

def go_close_example_3():
    import threading
    import queue
    import time

    def producer(ch):
        for i in range(5):
            ch.put(i)
            print(f'Produced: {i}')
            time.sleep(0.5)
        ch.put(None)  # Signal completion

    def consumer(ch):
        while True:
            item = ch.get()
            if item is None:
                break  # Exit on None
            print(f'Consumed: {item}')

    ch = queue.Queue()
    t1 = threading.Thread(target=producer, args=(ch,))
    t2 = threading.Thread(target=consumer, args=(ch,))
    t1.start()
    t2.start()
    t1.join()
    ch.put(None)  # Ensure consumer exits if not already done
    t2.join()
