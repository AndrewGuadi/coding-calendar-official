append

The 'append' function in Go is used to add elements to the end of a slice. It takes the existing slice and one or more elements to append. If the capacity of the slice is exceeded, a new underlying array will be allocated and the elements will be copied to this new array. This makes it a very versatile and commonly used function in Go for handling dynamic arrays.

package main

import (
	"fmt"
)

func main() {
	// Initializing a slice of strings
	slice := []string{"apple", "banana"}
	
	// Appending a single value
	slice = append(slice, "cherry")
	
	// Appending multiple values
	slice = append(slice, "date", "elderberry")
	
	// Printing the result
	fmt.Println(slice) // Output: [apple banana cherry date elderberry]
}
package main

import (
	"fmt"
)

func main() {
	// Creating a slice of integers
	slice := []int{1, 2, 3}
	
	// Appending a single integer
	slice = append(slice, 4)
	
	// Appending a slice of integers
	newSlice := []int{5, 6, 7}
	slice = append(slice, newSlice...)
	
	// Printing the result
	fmt.Println(slice) // Output: [1 2 3 4 5 6 7]
}
package main

import (
	"fmt"
)

func main() {
	// Initializing a slice of strings which represents names
	names := []string{"John", "Jane"}
	
	// Appending more names
