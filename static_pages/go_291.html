close

The 'close' method in Go is used to close a resource like a file, network connection, or any other I/O resource that requires cleanup. It ensures that all data is flushed, resources are released, and no further operations can be performed on the closed resource. This is crucial for avoiding resource leaks and ensuring that pending operations are completed.

import os

file_path = 'example.txt'

# Create and write to a file
with open(file_path, 'w') as file:
    file.write('This is a test file.')

# Closing the file automatically happens here due to the context manager

# Manually close example
file = open(file_path, 'r')
try:
    content = file.read()
    print(content)
finally:
    file.close()

import socket

# Create a socket connection
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

try:
    sock.connect(('localhost', 8080))
    sock.sendall(b'Hello, world')
    response = sock.recv(1024)
    print('Received', response)
finally:
    sock.close() # Ensure the socket is closed.

import sqlite3

# Connect to an SQLite database
connection = sqlite3.connect('example.db')

try:
    cursor = connection.cursor()
    cursor.execute('CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT)')
    cursor.execute('INSERT INTO users (name) VALUES (?)', ('Alice',))
    connection.commit()
finally:
    connection.close() # Always close the connection
