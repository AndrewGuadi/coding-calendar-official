close

In Go, the 'close' method is used to close a channel. Closing a channel indicates that no more values will be sent through it, which can be important for signaling to receivers that they can finish processing the channel. It’s essential to note that attempting to send values on a closed channel will cause a panic, whereas receiving from a closed channel will yield the zero value of the channel's type along with a boolean value indicating if it’s closed. It's a good practice to close channels to free resources and indicate that all sending is complete.

import time

values = []int([1, 2, 3, 4])
channel = []


def producer(channel):
    for value in values:
        channel.append(value)
        time.sleep(1)  # simulate work
    channel.close()


def consumer(channel):
    while True:
        if not channel:
            break  # terminate if channel is closed
        print(channel.pop(0))  # get the value and remove it

# Usage
ch = []
producer(ch)
consumer(ch)
import threading

def producer(channel):
    for i in range(5):
        channel.append(i)
    channel.close()


def consumer(channel):
    while True:
        if channel:
            print('Received:', channel.pop(0))
        else:
            break  # finish if channel is empty

# Usage
ch = []
threading.Thread(target=producer, args=(ch,)).start()
threading.Thread(target=consumer, args=(ch,)).start()
import queue
import threading


def producer(q):
    for i in range(5):
        q.put(i)
    q.put(None)  # put a sentinel to indicate closure


def consumer(q):
    while True:
        item = q.get()
        if item is None:
            break  # exit on sentinel
        print('Consumed:', item)

# Usage
q = queue.Queue()
producer_thread = threading.Thread(target=producer, args=(q,))
consumer_thread = threading.Thread(target=consumer, args=(q,))
producer_thread.start()
consumer_thread.start()
producer_thread.join()
q.put(None)  # close the queue
