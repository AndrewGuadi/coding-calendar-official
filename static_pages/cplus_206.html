std::vector::push_back

The 'push_back' method in C++ is used to add a new element to the end of a vector. This method allows the vector to dynamically increase its size, accommodating new elements as they are added. The new element is placed at the position equal to the current size of the vector, effectively increasing the size of the vector by one. This operation has an amortized time complexity of O(1), meaning that while it may take O(n) time in certain situations (e.g., when the vector needs to allocate more memory), on average, it operates in constant time as new elements are added.


#include 
#include 

int main() {
    std::vector myVector;
    myVector.push_back("First");
    myVector.push_back("Second");
    myVector.push_back("Third");

    for(const auto& item : myVector) {
        std::cout << item << std::endl;  // Prints 'First', 'Second', 'Third'
    }
    return 0;
}

int main() {
    std::vector names;
    names.push_back("Alice");
    names.push_back("Bob");
    names.push_back("Charlie");

    // Display the names stored in the vector
    for (size_t i = 0; i < names.size(); ++i) {
        std::cout << "Name " << (i + 1) << ": " << names[i] << std::endl;  // Outputs 'Name 1: Alice', etc.
    }
    return 0;
}

int main() {
    std::vector fruits;
    fruits.push_back("Apple");
    fruits.push_back("Banana");
    fruits.push_back("Cherry");

    // Convert and display fruits in uppercase
    for(const auto& fruit : fruits) {
        std::string upper_fruit;
        for(char c : fruit) {
            upper_fruit += toupper(c);
        }
        std::cout << upper_fruit << std::endl;  // Outputs 'APPLE', 'BANANA', 'CHERRY'
    }
    return 0;
}
