std::vector::push_back

The 'std::vector::push_back' method is used to add a new element to the end of a vector in C++. This operation increases the size of the vector by one, and the new element is initialized to the value provided. If the capacity of the vector needs to be increased to accommodate the new element, a new memory allocation is performed, and all existing elements are preserved in the new location.

```cpp
#include 
#include 

int main() {
    std::vector v;  // Create an empty vector of strings
    v.push_back("Hello, World!");  // Add a string to the vector
    for(const auto &element : v) {
        std::cout << element << std::endl;  // Output: Hello, World!
    }
    return 0;
}
```
```cpp
#include 
#include 

int main() {
    std::vector v;
    v.push_back("Python");  // Push string "Python"
    v.push_back("Java");    // Push string "Java"
    v.push_back("C++");     // Push string "C++"

    std::cout << "Languages:
";
    for(const auto &element : v) {
        std::cout << element << std::endl;  // Output: Python, Java, C++
    }
    return 0;
}
```
```cpp
#include 
#include 

int main() {
    std::vector v;
    std::vector new_languages = {"JavaScript", "Go", "Rust"};

    for(const auto &lang : new_languages) {
        v.push_back(lang);  // Push each language into the vector
    }

    std::cout << "Added Languages:
";
    for(const auto &element : v) {
        std::cout << element << std::endl;  // Output: JavaScript, Go, Rust
    }
    return 0;
}
```
