Add

The 'Add' method in C# is typically used to add an element to a collection, such as a list or a dictionary. In the context of a List, it appends the specified object to the end of the list. In the context of a Dictionary, it adds a new key-value pair. This method is essential for modifying collections dynamically at runtime, allowing developers to maintain and manipulate data structures efficiently.

using System;
using System.Collections.Generic;

public class Program
{
    public static void Main()
    {
        // Example of using Add with List
        List fruits = new List();
        fruits.Add("apple");
        fruits.Add("banana");
        fruits.Add("cherry");

        Console.WriteLine("List of fruits:");
        foreach (var fruit in fruits)
        {
            Console.WriteLine(fruit);
        }
    }
}
using System;
using System.Collections.Generic;

public class Program
{
    public static void Main()
    {
        // Example of using Add with Dictionary
        Dictionary capitals = new Dictionary();
        capitals.Add("USA", "Washington, D.C.");
        capitals.Add("France", "Paris");
        capitals.Add("Germany", "Berlin");

        Console.WriteLine("Capitals:");
        foreach (var kvp in capitals)
        {
            Console.WriteLine($"The capital of {kvp.Key} is {kvp.Value}");
        }
    }
}
using System;
using System.Collections.Generic;

public class Program
{
    public static void Main()
    {
        // Example of using Add with custom class
        List people = new List();
        people.Add(new Person { Name = "Alice", Age = 30 });
        people.Add(new Person { Name = "Bob", Age = 25 });

        Console.WriteLine("List of people:");
        foreach (var person in people)
        {
            Console.WriteLine($"Name: {person.Name}, Age: {person.Age}");
        }
    }
}

public class Person
{
    public string Name { get; set; }
    public int Age { get; set; }
}
