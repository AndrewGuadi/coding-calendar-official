close

In Go, the 'close' method is utilized to close channels. Closing a channel indicates that no more values will be sent on it, which is especially important for signaling to receiver goroutines that they should stop waiting for more data. When a channel is closed, any attempt to send data on that channel will cause a panic. However, receiving from a closed channel will not cause a panic; instead, it returns the zero value for the channel's type. It is a common practice to use a 'done' channel to notify goroutines when to stop processing.

import time

func main() {
    done := make(chan struct{})

    go func() {
        time.Sleep(2 * time.Second)
        close(done)
    }()

    <-done // wait for signal to stop
    fmt.Println("Goroutine finished.")
}
func main() {
    jobs := make(chan int, 5)
    done := make(chan bool)

    go func() {
        for job := range jobs {
            fmt.Println("Processing job", job)
        }
        done <- true
    }()

    for i := 1; i <= 5; i++ {
        jobs <- i
    }
    close(jobs) // Close jobs channel

    <-done // Wait for goroutine to finish
    fmt.Println("All jobs processed.")
}
func main() {
    data := make(chan string)
    go func() {
        data <- "Hello"
        data <- "World"
        close(data)
    }()

    for msg := range data { // Receiving from the channel
        fmt.Println(msg)
    }
    fmt.Println("Channel closed.")
}
