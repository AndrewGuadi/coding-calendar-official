abs

The `abs` method in Java is used to obtain the absolute value of a given number. This method is available in the `java.lang.Math` class and can accept various numeric types, including integers, floats, and doubles. The absolute value of a number is its non-negative value, meaning that any negative number will be converted to its positive counterpart. This method is particularly useful in mathematical calculations where the sign of a number does not matter.

import java.util.Scanner;

public class AbsoluteValueExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int number = scanner.nextInt();
        int absoluteValue = Math.abs(number);
        System.out.println("The absolute value of " + number + " is: " + absoluteValue);
        scanner.close();
    }
}
public class CalculateDistance {
    public static void main(String[] args) {
        double x1 = -5.5;
        double x2 = 3.2;
        double distance = Math.abs(x1 - x2);
        System.out.println("The distance between the points is: " + distance);
    }
}
public class AbsoluteValueArrays {
    public static void main(String[] args) {
        double[] numbers = {-2.3, 4.5, -1.5, 6.7};
        for (double num : numbers) {
            System.out.println("The absolute value of " + num + " is: " + Math.abs(num));
        }
    }
}
