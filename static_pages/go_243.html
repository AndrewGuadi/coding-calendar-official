close

In Go, the 'close' function is primarily used to close channels, which signals to other goroutines that no more values will be sent on that channel. Closing a channel is essential for avoiding deadlocks when you are done sending messages and allows the receiving goroutines to stop waiting for further incoming messages. Additionally, any receives on a closed channel will still immediately return the zero value of the channel's type, along with a boolean that indicates whether the channel is closed.

Go
package main

import (
	"fmt"
)

func main() {
	ch := make(chan string)

	go func() {
		ch <- "Hello"
		close(ch)
	}()

	msg, ok := <-ch
	if ok {
		fmt.Println(msg) // Output: Hello
	} else {
		fmt.Println("Channel is closed")
	}
}
Go
package main

import (
	"fmt"
)

func main() {
	ch := make(chan int)

	go func() {
		for i := 0; i < 5; i++ {
			ch <- i
		}
		close(ch)
	}()

	for num := range ch {
		fmt.Println(num) // Output: 0 to 4
	}
}
Go
package main

import (
	"fmt"
)

type Worker struct {
	ID int
}

func main() {
	ch := make(chan Worker)

	go func() {
		for i := 0; i < 3; i++ {
			ch <- Worker{ID: i}
		}
		close(ch)
	}()

	for worker := range ch {
		fmt.Println("Worker ID:", worker.ID) // Output: Worker ID: 0 to Worker ID: 2
	}
}
