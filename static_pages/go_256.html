close

In Go, the 'close' function is used to close a channel, indicating that no more values will be sent on it. Closing a channel is important in concurrent programming, as it helps to signal to receiving goroutines that there are no further values to be read from the channel. Attempting to send values on a closed channel will cause a panic, while receiving from a closed channel will return the zero value of the channel's type and a boolean indicating whether the channel is open.

def channel_example():
    import threading
    import queue

    def worker(ch):
        while True:
            item = ch.get()
            if item is None:
                break  # Exit the loop if None is received
            print(f'Processing {item}')

    ch = queue.Queue()  # Create a FIFO queue
    threading.Thread(target=worker, args=(ch,), daemon=True).start()  # Start worker thread

    for i in range(5):
        ch.put(i)  # Send data to channel

    ch.put(None)  # Signal the worker to stop

def close_channel_example():
    import threading
    import queue
    import time

    def producer(ch):
        for i in range(5):
            print(f'Sending {i}')
            ch.put(i)  # Send values to channel
            time.sleep(1)
        ch.close()  # Close the channel after sending values

    def consumer(ch):
        for item in iter(ch.get, None):  # Receive until the channel is closed
            print(f'Received {item}')

    ch = queue.Queue(maxsize=5)
    threading.Thread(target=producer, args=(ch,), daemon=True).start()  # Start producer thread
    threading.Thread(target=consumer, args=(ch,), daemon=True).start()  # Start consumer thread

def read_closed_channel_example():
    import threading
    import queue
    import time

    def producer(ch):
        for i in range(5):
            print(f'Sending {i}')
            ch.put(i)  # Send values to channel
            time.sleep(1)
        ch.close()  # Close the channel after sending values

    def consumer(ch):
        while True:
            try:
                item = ch.get(timeout=2)  # Timeout to avoid blocking
                print(f'Received {item}')
            except queue.Empty:
                print('No more items in channel.')
                break  # Exit if no items and channel is closed

    ch = queue.Queue()
    threading.Thread(target=producer, args=(ch,), daemon=True).start()  # Start producer thread
    time.sleep(0.1)  # Allow producer to start
    consumer(ch)
