abs

The 'abs' method in Java is used to return the absolute value of a given number. It can handle various numeric types including int, long, float, and double. This method is useful when you need to ensure that a number is non-negative, especially in mathematical computations where the distance or magnitude of a value is important.

import java.util.Scanner; public class AbsExample1 { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.print("Enter a number: "); int num = sc.nextInt(); System.out.println("The absolute value is: " + Math.abs(num)); } }
public class AbsExample2 { public static void main(String[] args) { double negativeDouble = -45.67; double absoluteValue = Math.abs(negativeDouble); System.out.println("The absolute value of " + negativeDouble + " is: " + absoluteValue); } }
public class AbsExample3 { public static void main(String[] args) { long longNum = -100000L; long absValue = Math.abs(longNum); System.out.println("The absolute value of " + longNum + " is: " + absValue); } }
