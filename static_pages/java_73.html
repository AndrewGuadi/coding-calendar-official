abs

The 'abs' method in Java is used to calculate the absolute value of a number. Absolute value is defined as the distance of a number from zero on the number line, regardless of direction. This method can be applied to integers, long integers, floating-point numbers, and double precision numbers. It is part of the 'java.lang.Math' class and returns the absolute value of the given argument. If the argument is already a positive number or zero, it returns the number itself; if it is negative, it returns its positive equivalent.

import java.util.Scanner;

public class AbsExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter an integer:");
        int number = scanner.nextInt();

        int absoluteValue = Math.abs(number);
        System.out.println("The absolute value of " + number + " is " + absoluteValue);
    }
}
# Given input: -42
# Output: The absolute value of -42 is 42

import java.util.Scanner;

public class AbsDoubleExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a floating-point number:");
        double number = scanner.nextDouble();

        double absoluteValue = Math.abs(number);
        System.out.println("The absolute value of " + number + " is " + absoluteValue);
    }
}
# Given input: -3.14
# Output: The absolute value of -3.14 is 3.14

public class AbsArrayExample {
    public static void main(String[] args) {
        int[] numbers = {-1, 2, -3, 4};
        System.out.println("Absolute values of the array:");
        for (int number : numbers) {
            int absoluteValue = Math.abs(number);
            System.out.print(absoluteValue + " ");
        }
    }
}
# Output: Absolute values of the array:
# 1 2 3 4
