Add

The 'Add' method is used to add an element to a collection such as a list, dictionary, or any other collection that implements the Add functionality in C#. This method modifies the collection by inserting the specified element at the end or at a specific index, depending on the type of collection. It is commonly used to dynamically manage collections of data during runtime by enabling the addition of new elements.

def add_to_list(my_list, item):
    my_list.append(item)
    return my_list

items = []
items = add_to_list(items, 'apple')  # items = ['apple']
items = add_to_list(items, 'banana') # items = ['apple', 'banana']
def add_to_dict(my_dict, key, value):
    my_dict[key] = value
    return my_dict

data = {}
data = add_to_dict(data, 'name', 'John Doe')  # data = {'name': 'John Doe'}
data = add_to_dict(data, 'age', 30)  # data = {'name': 'John Doe', 'age': 30}
def add_to_set(my_set, value):
    my_set.add(value)
    return my_set

unique_values = set()
unique_values = add_to_set(unique_values, 'value1')  # unique_values = {'value1'}
unique_values = add_to_set(unique_values, 'value2')  # unique_values = {'value1', 'value2'}
