close

In Go, the 'close' method is commonly used to close channels that have been opened for communication between goroutines. Closing a channel indicates that no more values will be sent on it. This is important for properly synchronizing goroutines and avoiding deadlocks. Attempting to send a value on a closed channel will cause a panic, so itâ€™s essential to manage the lifecycle of channels carefully.

import time
import threading

def worker(ch):
    for i in range(5):
        ch.append(i)
        print('Sent:', i)
        time.sleep(0.5)
    # Close the channel after sending all messages
    ch.close()

ch = []
thread = threading.Thread(target=worker, args=(ch,))
thread.start()

# Simulating receiving messages
while True:
    if not ch:
        break
    try:
        value = ch.pop(0)
        print('Received:', value)
    except IndexError:
        break
    time.sleep(0.2)

def producer(ch):
    for i in range(5):
        ch.append(i)
        print('Produced:', i)
        time.sleep(1)
    ch.close()

ch = []
producer_thread = threading.Thread(target=producer, args=(ch,))
producer_thread.start()

while True:
    if not ch:
        break
    try:
        value = ch.pop(0)
        print('Consumed:', value)
    except IndexError:
        break
    time.sleep(0.5)
import time
import threading

def main():
    messages = []

    def send_messages():
        for msg in ['A', 'B', 'C']:
            messages.append(msg)
            print(f'Sent: {msg}')
            time.sleep(1)
        messages.append(None)  # Signal closure here

    sender_thread = threading.Thread(target=send_messages)
    sender_thread.start()

    while True:
        if messages:
            message = messages.pop(0)
            if message is None:
                print('Channel closed.')
                break
            print(f'Received: {message}')
        else:
            time.sleep(0.5)
    sender_thread.join()

main()
