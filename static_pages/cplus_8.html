std::vector::push_back

The 'std::vector::push_back' method is a member function of the C++ Standard Library's vector class, which allows you to add an element to the end of a vector. This method automatically increases the size of the vector by one, storing the new element at the last position. If the vector's capacity is exceeded, it will allocate additional memory to accommodate the new element. This operation has an average time complexity of O(1).

# Example demonstrating basic usage of std::vector::push_back

#include 
#include 

int main() {
    std::vector myVector;
    myVector.push_back("Python"); // Adding first element
    myVector.push_back("Java");   // Adding second element
    myVector.push_back("C++");    // Adding third element

    // Printing the elements of the vector
    for (const auto& lang : myVector) {
        std::cout << lang << std::endl;
    }
    return 0;
}
# Example showing how std::vector::push_back can be used in a loop

#include 
#include 

int main() {
    std::vector languages;
    std::vector langList = {"Python", "Java", "C++", "JavaScript"};

    // Using push_back in a loop to add all languages to the vector
    for (const auto& lang : langList) {
        languages.push_back(lang);
    }

    // Displaying all added languages
    for (const auto& language : languages) {
        std::cout << language << std::endl;
    }
    return 0;
}
# Example of using std::vector::push_back with conditionals

#include 
#include 

int main() {
    std::vector selectedLanguages;
    std::vector availableLanguages = {"Python", "Java", "C++", "Ruby"};

    // Conditionally adding languages to the vector based on a selection criteria
    for (const auto& lang : availableLanguages) {
        if (lang != "C++") { // Exclude C++
            selectedLanguages.push_back(lang);
        }
    }

    // Displaying selected languages
    for (const auto& language : selectedLanguages) {
        std::cout << language << std::endl;
    }
    return 0;
}
