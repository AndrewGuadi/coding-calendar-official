functools.lru_cache

The functools.lru_cache decorator is part of the functools module, which stands for "Least Recently Used Cache". It stores the results of function calls and reuses them when the same inputs occur again, avoiding redundant computations. This is particularly useful for functions with expensive or repetitive computations.

Example 1: Fibonacci Sequence

from functools import lru_cache

@lru_cache(maxsize=None)
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# Testing the function
for i in range(10):
    print(f"Fibonacci({i}) = {fibonacci(i)}")

Example 2: Factorial Calculation

from functools import lru_cache

@lru_cache(maxsize=128)
def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n-1)

# Testing the function
print(f"Factorial(5) = {factorial(5)}")
print(f"Factorial(10) = {factorial(10)}")

Example 3: Prime Check

from functools import lru_cache

@lru_cache(maxsize=256)
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

# Testing the function
print(f"Is 17 prime? {is_prime(17)}")
print(f"Is 18 prime? {is_prime(18)}")
